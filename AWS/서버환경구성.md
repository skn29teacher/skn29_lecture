# Gunicorn 설정
pip install gunicorn

# Gunicorn 단독 테스트 실행(정상동작 확인)
gunicorn --bind 0.0.0.0:8000 config.wsgi:application

# 서버가 재 부팅될때 자동으로 Gunicorn을 실행하도록 systemd 등록
sudo vim /etc/systemd/system/gunicorn.service
```
[Unit]
Description=gunicorn daemon for Django
After=network.target

[Service]
User=ubuntu
Group=www-data
WorkingDirectory=/home/ubuntu/myproject
ExecStart=/home/ubuntu/myproject/venv/bin/gunicorn \
          --workers 3 \
          --bind unix:/home/ubuntu/myproject/gunicorn.sock \
          config.wsgi:application

[Install]
WantedBy=multi-user.target
```

# 서비스 시작 및 등록
```
sudo systemctl daemon-reload          # 서비스 파일 변경 반영
sudo systemctl start gunicorn         # 서비스 시작
sudo systemctl enable gunicorn        # 서버 재부팅 시 자동 시작 등록
sudo systemctl status gunicorn        # 상태 확인 (active/running이면 정상)
```

# nginx 설치 및 프록시 설정
sudo apt install -y nginx

# nginx 사이트 설정 파일 생성
sudo nano /etc/nginx/sites-available/myproject
```
server {
    listen 80;
    server_name <EC2 퍼블릭 IP>;

    location = /favicon.ico { access_log off; log_not_found off; }

    location /static/ {
        alias /home/ubuntu/myproject/static/;
    }

    location / {
        include proxy_params;
        proxy_pass http://unix:/home/ubuntu/myproject/gunicorn.sock;
    }
}
```

# config/settings.py에 추가
```
import os
STATIC_ROOT = os.path.join(BASE_DIR, 'static/')
```

```
python manage.py collectstatic --noinput

# Nginx 사이트 활성화
sudo ln -s /etc/nginx/sites-available/myproject /etc/nginx/sites-enabled

# 기본 사이트 비활성화 (충돌 방지)
sudo rm -f /etc/nginx/sites-enabled/default

# 설정 문법 검사
sudo nginx -t

# Nginx 재시작
sudo systemctl restart nginx
```

# 트러블슈팅


| 증상 | 원인 후보 | 점검 명령어 |
|------|-----------|--------------| 
| 502 Bad Gateway | Gunicorn이 안 떠 있음 / 소켓 파일 없음 | `sudo systemctl status gunicorn` → `sudo journalctl -u gunicorn` |
| 502 Bad Gateway | 소켓 파일 경로 불일치 | Nginx 설정의 `proxy_pass` 경로와 gunicorn.service의 `--bind` 경로가 같은지 확인 |
| 403 Forbidden | Nginx가 소켓 파일에 접근 불가 | `chmod 755 /home/ubuntu` 실행, `ls -la /home/ubuntu/gunicorn.sock` 확인 |
| 정적파일 깨짐(CSS 미적용) | `collectstatic` 미실행 | `python manage.py collectstatic --noinput` |
| 정적파일 404 | Nginx에서 `root` vs `alias` 혼동 | Nginx 설정에서 `alias /home/ubuntu/myproject/static/;` 확인 |
| 변경사항 반영 안됨 | 서비스 재시작 누락 | `sudo systemctl restart gunicorn nginx` |
