# 패키지 목록 업데이트 및 업그레이드
sudo apt update && sudo apt upgrade -y

# Python, venv, pip, git, 기타 빌드 도구 설치
sudo apt install -y python3-pip python3-venv python3-dev git build-essential libpq-dev

# 작업 디렉토리 생성 및 이동
mkdir ~/myproject && cd ~/myproject

# 가상환경 생성 및 활성화
python3 -m venv venv
source venv/bin/activate

# Django 및 필수 패키지 설치
pip install django gunicorn

# Django 프로젝트 생성 (현재 디렉토리에)
django-admin startproject config .

# 초기 데이터베이스 테이블 생성
python manage.py migrate

config/settings.py 수정 (nano 또는 vi 편집기 또는 vim 사용):

nano config/settings.py
# ALLOWED_HOSTS를 찾아 수정
ALLOWED_HOSTS = ['<EC2 퍼블릭 IP>', 'localhost', '127.0.0.1']

# 개발 서버 실행:
python manage.py runserver 0.0.0.0:8000

# Gunicorn 설정
pip install gunicorn

# Gunicorn 단독 테스트 실행(정상동작 확인)
gunicorn --bind 0.0.0.0:8000 config.wsgi:application

# 서버가 재 부팅될때 자동으로 Gunicorn을 실행하도록 systemd 등록
sudo vim /etc/systemd/system/gunicorn.service
```
[Unit]
Description=gunicorn daemon for Django
After=network.target

[Service]
User=ubuntu
Group=www-data
WorkingDirectory=/home/ubuntu/myproject
ExecStart=/home/ubuntu/myproject/venv/bin/gunicorn \
          --workers 3 \
          --bind unix:/home/ubuntu/myproject/gunicorn.sock \
          config.wsgi:application

[Install]
WantedBy=multi-user.target
```

# 서비스 시작 및 등록
```
sudo systemctl daemon-reload          # 서비스 파일 변경 반영
sudo systemctl start gunicorn         # 서비스 시작
sudo systemctl enable gunicorn        # 서버 재부팅 시 자동 시작 등록
sudo systemctl status gunicorn        # 상태 확인 (active/running이면 정상)
```
